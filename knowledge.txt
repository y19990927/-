git init:                               初始化仓库
git add:                                把文件添加到暂存区
git commit -m:                          提交文件 -m 后面是注释
git status:                             查看仓库状态
git diff:                               查看文件修改了哪些内容
git log:                                查看历史记录
git reflog:                             查看历史记录的版本号
cat:                                    查看文件内容
git checkout -- :                       撤销修改
git rm:                                 删除文件
git renite add origin 路径:             关联远程库
git push -u(-u第一次使用) origin 分支:  把当前分支推送到远程库
git clone:                              克隆
git checkout -b 分支:                   创建并切换分支
git branch:                             查看所有分支
git checkout 分支;                      切换分支
git merge 分支:                         合并分支
git branch 分支:                        创建分支
